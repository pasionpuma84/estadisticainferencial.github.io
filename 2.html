Gráfica Circular

    genero_counts = df_marketing['Genero'].value_counts()
    print(genero_counts)

    ax = genero_counts.plot(kind='pie',
    colors=['#a7d2f1', '#f3b0e1', '#ff6347', '#ffd700'], # Puedes personalizar los colores
    figsize=(10, 6),
    autopct='%1.1f%%', # Muestra los porcentajes
    startangle=90, # Empieza el gráfico desde un ángulo fijo
    wedgeprops={'edgecolor': 'white', 'linewidth': 1.5},
    textprops={'color': 'black','fontsize': 18}
    ) # Bordes de las secciones

    # Personalizar la gráfica
    plt.title('Porcentaje de Genero', fontsize=16, fontweight='bold') # Título más grande y en negrita
    plt.ylabel('') # Eliminar la etiqueta de "Genero" en el eje Y
    plt.xticks(rotation=0) # Asegurar que los nombres de las categorías se vean bien

    # Ajustar el layout para no recortar

    plt.tight_layout()

    # Mostrar la gráfica
    plt.show()

Gráfica de barras horizontales

    edad_counts = df_marketing['Edad'].value_counts()
    print(edad_counts)

    # Crear la gráfica de barras
    ax=edad_counts.plot(kind='barh', color='Skyblue', figsize=(10, 6))

    # Personalizar la gráfica
    plt.title('Frecuencia de Edades')
    plt.xlabel('Frecuenca')
    plt.ylabel('Edad')
    plt.xticks(rotation=0)
    for i, valor in enumerate(edad_counts.values):
        ax.text(valor + 0.5, i, str(valor), ha='left', va='center', fontsize=10, color='blue')
    plt.tight_layout()
    plt.show()
